----INIT
USE ROLE SECURITYADMIN;
CREATE OR REPLACE ROLE NEW_GROUP;
GRANT ROLE NEW_GROUP TO USER "BOAZ";

-------
CREATE OR REPLACE ROLE AC_NEW_GROUP;
CREATE OR REPLACE ROLE DWH_FINANCE;
GRANT USAGE ON DATABASE DWH TO ROLE DWH_FINANCE;
GRANT USAGE ON SCHEMA DWH.FINANCE TO ROLE DWH_FINANCE;
GRANT SELECT  ON DWH.FINANCE.FACT_STAGE_6 IN SCHEMA DWH.FINANCE TO ROLE DWH_FINANCE;
USE ROLE SYSADMIN;
GRANT USAGE, OPERATE ON WAREHOUSE DEV_DWH_WH TO ROLE DWH_FINANCE;

--HIERARCHY
GRANT ROLE DWH_FINANCE TO ROLE AC_NEW_GROUP;
GRANT ROLE AC_NEW_GROUP TO ROLE NEW_GROUP;

------CLEANUP
USE ROLE SECURITYADMIN;
DROP ROLE IF EXISTS NEW_GROUP;
DROP ROLE IF EXISTS AC_NEW_GROUP;
DROP ROLE IF EXISTS DWH_FINANCE;
